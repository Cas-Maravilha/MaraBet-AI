## üìä M√©tricas e KPIs

### M√©tricas de Neg√≥cio

```python
class BusinessMetrics:
    def __init__(self):
        self.metrics = {
            'total_predictions': 0,
            'accurate_predictions': 0,
            'api_requests': 0,
            'error_count': 0,
            'uptime': 0
        }
    
    def calculate_accuracy(self):
        """Calcula taxa de acerto"""
        if self.metrics['total_predictions'] == 0:
            return 0
        return self.metrics['accurate_predictions'] / self.metrics['total_predictions']
    
    def calculate_uptime(self):
        """Calcula uptime do sistema"""
        total_time = time.time() - self.start_time
        downtime = self.metrics['error_count'] * 60  # 1 min por erro
        return (total_time - downtime) / total_time
    
    def generate_report(self):
        """Gera relat√≥rio de m√©tricas"""
        return {
            'accuracy': self.calculate_accuracy(),
            'uptime': self.calculate_uptime(),
            'total_predictions': self.metrics['total_predictions'],
            'api_requests': self.metrics['api_requests'],
            'error_rate': self.metrics['error_count'] / max(self.metrics['api_requests'], 1)
        }
```
